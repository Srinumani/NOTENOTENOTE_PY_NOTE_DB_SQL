WITH total_sales AS (
    SELECT 
        emp_id,
        SUM(sale_amount) AS total_sales_emp
    FROM sales
    GROUP BY emp_id
),

avg_total_sales AS (
    SELECT AVG(total_sales_emp) AS avg_total
    FROM total_sales
)

SELECT 
    e.name,
    t.total_sales_emp
FROM total_sales t
JOIN avg_total_sales a
    ON t.total_sales_emp > a.avg_total
JOIN employees e
    ON e.emp_id = t.emp_id;
